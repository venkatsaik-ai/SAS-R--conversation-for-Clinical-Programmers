PROC SORT – SAS vs R

PROC SORT with NODUP

SAS PROC SORT NODUP

data example;
input id age;
datalines;
1 25
2 30
2 30
3 28
3 28
;
run;

proc sort data=example nodup;
by id age;
run;
Output Dataset ID AGE
1 25
2 30
3 28

R distinct()

library(dplyr)

example <- data.frame(
id=c(1,2,2,3,3),
age=c(25,30,30,28,28)
)

distinct(example)
id age
1 25
2 30
3 28
PROC SORT with NODUPKEY

SAS PROC SORT NODUPKEY

data example;
input id age;
datalines;
1 25
2 30
2 35
3 28
3 40
;
run;

proc sort data=example nodupkey;
by id;
run;
Output Dataset ID AGE
1 25
2 30
3 28

R distinct by column

library(dplyr)

example <- data.frame(
id=c(1,2,2,3,3),
age=c(25,30,35,28,40)
)

distinct(example, id, .keep_all=TRUE)
id age
1 25
2 30
3 28
