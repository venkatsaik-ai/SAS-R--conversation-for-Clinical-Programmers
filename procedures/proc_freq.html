PROC FREQ – SAS vs R

Single Frequency Table

SAS PROC FREQ

data example;
input gender $;
datalines;
M
F
M
M
F
;
run;

proc freq data=example;
tables gender;
run;
M = 3
F = 2

R table()

gender <- c("M","F","M","M","F")

table(gender)
M = 3
F = 2
Cross Tabulation

SAS Crosstab

data example;
input gender $ treatment $;
datalines;
M A
F A
M B
F B
M A
;
run;

proc freq data=example;
tables gender*treatment;
run;
Treatment

Gender   A   B
M          2   1
F          1   1
