PROC TRANSPOSE – SAS vs R

Basic Transpose

SAS PROC TRANSPOSE

data example;
input id math science;
datalines;
1 80 85
2 75 90
3 88 92
;
run;

proc transpose data=example
out=transposed;
var math science;
run;
Output Dataset NAME COL1 COL2 COL3
math 80 75 88
science 85 90 92

R transpose using t()

example <- data.frame(
id=c(1,2,3),
math=c(80,75,88),
science=c(85,90,92)
)

t(example[,2:3])
[,1] [,2] [,3]
math 80 75 88
science 85 90 92
Transpose using BY and ID

SAS PROC TRANSPOSE with BY and ID

data example;
input id subject $ score;
datalines;
1 Math 80
1 Science 85
2 Math 75
2 Science 90
;
run;

proc sort data=example;
by id;
run;

proc transpose data=example
out=transposed;
by id;
id subject;
var score;
run;
Output Dataset ID Math Science
1 80 85
2 75 90

R pivot_wider()

library(tidyr)

example <- data.frame(
id=c(1,1,2,2),
subject=c("Math","Science","Math","Science"),
score=c(80,85,75,90)
)

pivot_wider(example,
names_from=subject,
values_from=score)
id Math Science
1 80 85
2 75 90
